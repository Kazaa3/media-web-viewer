    async function runVideoDiagnosticsUI() {
        const btn = document.getElementById('run-video-diag-btn');
        const statusBox = document.getElementById('video-diag-status');
        const emptyState = document.getElementById('video-diag-empty-state');
        const resultsContainer = document.getElementById('video-diag-results-container');
        const targetInfo = document.getElementById('video-diag-target-info');
        
        btn.disabled = true;
        btn.innerText = 'Probing...';
        emptyState.style.display = 'none';
        
        try {
            const res = await eel.run_video_transcode_diagnostic()();
            statusBox.style.display = 'block';
            
            if (res.status === 'complete') {
                targetInfo.innerText = `PROBE TARGET: ${res.target}`;
                resultsContainer.innerHTML = res.results.map(r => `
                    

                        

                            ${r.name} Pipeline
                            ${r.status}
                        

                        
${r.details}

                    

                `).join('');
            } else {
                targetInfo.innerText = `ERROR: ${res.error || 'Unknown failure'}`;
                resultsContainer.innerHTML = '';
            }
        } catch (e) {
            console.error(e);
            targetInfo.innerText = `CRITICAL FAILURE: ${e.message}`;
        } finally {
            btn.disabled = false;
            btn.innerText = 'Run Pipeline Probe';
        }
    }
